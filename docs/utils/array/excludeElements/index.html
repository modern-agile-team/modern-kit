import { excludeElements } from '@modern-kit/utils';

  { name: 'kim', address: { city: 'Seoul' } },
  { name: 'lee', address: { city: 'NewYork' } },

excludeElements(array, excluded, (item) => item.name); // { name: 'lee', address: { city: 'NewYork' } }
