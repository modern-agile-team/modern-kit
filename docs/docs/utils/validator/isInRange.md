## Interface
```ts title="typescript"

interface IsInRangeProps {
  value: number;
  min: number;
  max: number;
  equalOptions?: {
    min?: boolean;
    max?: boolean;
  };
}

const isInRange = ({
  value,
  min,
  max,
  equalOptions = {
    min: true,
    max: false,
  },
}: IsInRangeProps) => boolean;
```
