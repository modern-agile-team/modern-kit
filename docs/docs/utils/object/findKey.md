## Interface
```ts title="typescript"
function findKey<T extends Record<PropertyKey, any>>(
  obj: T,
  predicate: (predicateData: {
    value: T[keyof T];
    key: keyof T;
    obj: T;
  }) => boolean
): string | undefined;
```

## Usage
```ts title="typescript"
import { findKey } from '@modern-kit/utils';

const obj = {
   bike: { active: true },
   plane: { active: true },
   car: { active: false },
};

findKey(obj, (item) => item.active); // 'bike'
```
