## Interface
```ts title="typescript"
type SetIntervalParameters = Parameters<typeof setInterval>;

type IntervalOptions =
  | number
  | { delay: SetIntervalParameters[1]; enabled?: boolean };

const useInterval: (
  callback: SetIntervalParameters[0],
  options: IntervalOptions
) => void;
```
