## Usage
```tsx title="typescript"
import { useForceUpdate } from '@modern-kit/react';

const Example = () => {
  const forceUpdate = useForceUpdate();

  const handleForceUpdate = useCallback(() => {
    forceUpdate();
  }, [forceUpdate]);

  return (
    <div>
      <button onClick={handleForceUpdate}>Button</button>;
    </div>
  );
};
```
