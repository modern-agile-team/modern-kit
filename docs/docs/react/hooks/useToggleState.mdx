## Usage
```tsx title="typescript"
import { useToggleState } from '@modern-kit/react';

const Example = () => {
  const [value, toggle] = useToggleState('ON', 'OFF');

  return (
    <div>
      {value}
      <button onClick={toggle}>Toggle Button</button>
    </div>
  );
}
```
