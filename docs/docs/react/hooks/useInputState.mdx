## Interface
```ts title="typescript
function useInputState(initialValue: T): {
  value: T | undefined;
  onChange: (e: ChangeEvent<HTMLInputElement>) => void;
  reset: () => void;
}
```

## Usage
```tsx title="typescript"

import { useInputState } from '@modern-kit/react';

  const { value, onChange, reset } = useInputState<string>();

  return (
    <>
      <h1>{value}</h1>
      <input type="text" name="test" onChange={onChange} value={value} />
```

## Example

export const ExampleSingleInput = () => {
  const { value, onChange, reset } = useInputState<string>('');

  return (
    <>
      <h1>{value}</h1>
      <input type="text" name="test" onChange={onChange} value={value} />
      <button onClick={reset}>reset</button>
    </>
  );
};

export const ExampleMultiple = () => {
  const { value, onChange, reset } = useInputState<{
    title: string;
    description: string;
  }>({
    title: '',
    description: '',
  });

  return (
    <>
      <h1>{value.title}</h1>
      <p>{value.description}</p>
      <input type="text" name="title" onChange={onChange} value={value.title} />
      <input
        type="text"
        name="description"
        onChange={onChange}
        value={value.description}
      />
      <button onClick={reset}>reset</button>
    </>
  );
};
