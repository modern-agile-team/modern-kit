## Interface
```ts title="typescript"
function Portal({
  children,
  containerRef,
}: {
  children: React.ReactNode;
  containerRef?: React.RefObject<HTMLElement | null>;
}): JSX.Element;
```

## Usage
### Default

```tsx title="typescript"
import { Portal } from '@modern-kit/react'

const Example = () => {
  return (
    <Portal>
      <p>Example Portal</p>
    </Portal>
  );
};
```

### Container

```tsx title="typescript"
import { Portal } from '@modern-kit/react'

const Example = () => {
  const ref = useRef<HTMLDivElement | null>(null);

  return (
    <div>
      <Portal containerRef={ref}>
        <p>Example Portal</p>
      </Portal>

      <div id="outer" ref={ref} />
    </div>
  );
};
```

### Nested

```tsx title="typescript"
import { Portal } from '@modern-kit/react'

const Example = () => {
  return (
    <Portal>
      <p>Default Portal</p>
      <Portal>
        <p>Nested Portal1</p>
        <Portal>
          <p>Nested Portal2</p>
        </Portal>
      </Portal>
    </Portal>
  );
};
```
