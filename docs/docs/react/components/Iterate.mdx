## Interface
```ts title="typescript"
interface IterateProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => JSX.Element;
}

const Iterate: <T extends unknown>({
  items,
  renderItem,
}: IterateProps<T>) => JSX.Element;
```

## Usage
```tsx title="typescript"
import { Iterate } from '@modern-kit/react';

interface Item {
  id: number;
  name: string;
}

const Example = () => {
  const items: Item[] = [
    { id: 1, name: 'Gromit' },
    { id: 2, name: 'Dylan' },
    { id: 3, name: 'Minjae' },
  ];

  return (
    <Iterate
      items={items}
      renderItem={(item) => (
        <h3>
          id: {item.id}, name:{item.name}
        </h3>
      )}
    />
  );
};
```
